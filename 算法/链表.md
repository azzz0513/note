```Go
func reverseList(head *ListNode) *ListNode {
    if head == nil {
        return nil
    }
    node := head.Next
    tmp := head
    for node != nil {
        tmp.Next = node.Next
        node.Next = head
        head = node
        node = tmp.Next
    }
    return head
}
```

## 快慢指针

代码：
```Go
func hasCycle(head *ListNode) bool {
    fast, slow := head, head
    for fast != nil && fast.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
        if fast == slow {
            return true
        }
    }
    return false
}
```

func detectCycle(head *ListNode) *ListNode {
    slow, fast := head, head
    for fast != nil && fast.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
        if fast == slow {
            for head != slow {
                head = head.Next
                slow = slow.Next
            }
            return slow
        }
    }
    return nil
}
