```Go
func maxDepth(root *TreeNode) int {
    if root == nil {
        return 0
    }
    
    return max(maxDepth(root.Left), maxDepth(root.Right)) + 1
}
```

```Go
func maxDepth(root *TreeNode) int {
    ans := 0

    var f func(root *TreeNode, cnt int)
    f = func(root *TreeNode, cnt int) {
        if root == nil {
            return
        }
        
        cnt++
        if cnt > ans {
            ans = cnt
        }
        f(root.Left, cnt)
        f(root.Right, cnt)
    }
  
    f(root, 0)
    return ans
}
```
