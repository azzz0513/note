## 读取一行，按空格/逗号分割
- 读取一行，按空格分割
```Go
package main 

import ( 
	"bufio" 
	"fmt" 
	"os" 
	"strings" 
) 

func main() { 
	scanner := bufio.NewScanner(os.Stdin) 
	scanner.Scan() // 读一行 
	line := scanner.Text() 
	parts := strings.Fields(line) // 按空格分割，自动去空 
	fmt.Println(parts) 
}
```

- 读取一行，按逗号分割
```Go
scanner.Scan() 
line := scanner.Text() 
parts := strings.Split(line, ",")
```

## 读取数字（`int`/`[]int`）
- 读取单个整数
```Go
var x int 
fmt.Scan(&x)
```

- 读取一行多个整数
```Go
scanner := bufio.NewScanner(os.Stdin) 
scanner.Scan() 
parts := strings.Fields(scanner.Text()) 

nums := make([]int, len(parts)) 
for i := 0; i < len(parts); i++ { 
	fmt.Sscan(parts[i], &nums[i]) 
}
```

## 读取多行数据（循环读）
```Go
scanner := bufio.NewScanner(os.Stdin) 
for scanner.Scan() { 
	line := scanner.Text() 
	if line == "" { 
		break 
	} 
	// 处理 line 
}
```

# 万能模板
```Go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

var (
	scanner *bufio.Scanner
	writer  *bufio.Writer
)

func main() {
	// 快速读入 + 快速输出
	scanner = bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanLines) // 按行读
	writer = bufio.NewWriter(os.Stdout)
	defer writer.Flush() // 最后统一输出

	// 1. 单组用一次 NextLine()
	// 2. 多组用 for 循环读

	// 示例：多组测试用例
	// for scanner.Scan() {
	// 	line := scanner.Text()
	// 	if line == "" {
	// 		break
	// 	}
	// 	// 处理
	// }

	// 你的逻辑写这里
}

// 读取一行
func NextLine() string {
	scanner.Scan()
	return scanner.Text()
}

// 读取一行，转 []string（按空格分割）
func NextStrings() []string {
	return strings.Fields(NextLine())
}

// 读取一行，转 []int
func NextInts() []int {
	strs := NextStrings()
	res := make([]int, len(strs))
	for i := range strs {
		res[i], _ = strconv.Atoi(strs[i])
	}
	return res
}

// 读取单个整数
func NextInt() int {
	return NextInts()[0]
}

// 快速输出
func Print(a ...interface{}) {
	fmt.Fprint(writer, a...)
}

func Println(a ...interface{}) {
	fmt.Fprintln(writer, a...)
}

func Printf(format string, a ...interface{}) {
	fmt.Fprintf(writer, format, a...)
}
```


## 简版模板
```Go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	sc := bufio.NewScanner(os.Stdin)
	sc.Split(bufio.ScanLines)
	wr := bufio.NewWriter(os.Stdout)
	defer wr.Flush()

	// 读一行
	sc.Scan()
	parts := strings.Fields(sc.Text())

	// 转int
	n, _ := strconv.Atoi(parts[0])

	// 输出
	fmt.Fprintln(wr, n)
}
```
