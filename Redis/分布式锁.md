## 基于Redis的分布式锁
实现分布式锁时需要实现的两个基本方法：
- 获取锁：
	- 互斥：确保只能有一个线程获取锁
```
# 添加锁，利用setnx的互斥特性
SETNX lock thread1

# 添加锁过期时间，避免服务宕机引起的死锁
EXPIRE lock 10

或

# 添加锁，NX是互斥，EX是设置超时时间
SET lock thread1 NX EX 10
```

- 释放锁：
	- 手动释放
	- 超时释放：获取锁时添加一个超时时间
```
# 释放锁，删除即可
DEL lock
```

### Redis的Lua脚本
Redis提供了Lua脚本功能，在一个脚本中编写多条redis命令，确保多条命令执行时的原子性。
Redis提供的调用函数：
```
# 执行redis命令
redis.call('命令名称', 'key', '其他参数', ...)
```

调用脚本：使用`EVAL`命令
如果脚本中的key，value不想写死，可以作为参数传递。key类型参数会放入KEYS数组，其他参数会放入ARGV数组，在脚本中可以从KEYS和ARGV数组获取这些参数
```Go
var unlockScript = `
if redis.call("get", KEYS[1]) == ARGV[1] then
    return redis.call("del", KEYS[1])
else
    return 0
end
`
// 使用 Lua 脚本原子地检查令牌并删除键
res, err := l.client.Eval(ctx, unlockScript, []string{l.key}, l.token).Result()
if err != nil {
	return false, err
}
```

## 代码实现
```Go
type Locker struct {
    client *redis.Client
    key    string
    token  string
    ttl    time.Duration
}

  

// NewLocker 创建一个新的 Locker 实例。
func NewLocker(client *redis.Client, key string, ttl time.Duration) *Locker {
    return &Locker{
        client: client,
        key: fmt.Sprintf("lock:%s", key),
        ttl: ttl,
    }
}

  

// genToken 生成一个随机的16字节令牌，用于锁定。
func genToken(nbytes int) (string, error) {
    b := make([]byte, nbytes)
    if _, err := rand.Read(b); err != nil {
        return "", err
    }
    return hex.EncodeToString(b), nil
}

  

// TryLock 尝试获取锁，返回是否成功和可能的错误。

func (l *Locker) TryLock(ctx context.Context) (bool, error) {
    if l.client == nil {
        return false, errors.New("nil redis client")
    }
    // 生成一个随机令牌
    token, err := genToken(16)
    if err != nil {
        return false, err
    }

    // 使用 SET NX PX 原子操作通过 SetNX 包装器
    ok, err := l.client.SetNX(ctx, l.key, token, l.ttl).Result()
    if err != nil {
        return false, err
    }
    if !ok {
        return false, nil
    }
    l.token = token
    return true, nil
}

  

// 解锁脚本，只有当键的值与令牌匹配时才删除键
var unlockScript = `
if redis.call("get", KEYS[1]) == ARGV[1] then
    return redis.call("del", KEYS[1])
else
    return 0
end
`

  

// Unlock 释放锁，返回是否成功和可能的错误。
// 注意：只有持有锁的实例才能释放锁。
func (l *Locker) Unlock(ctx context.Context) (bool, error) {
    if l.client == nil {
        return false, errors.New("nil redis client")
    }
    if l.token == "" {
        return false, errors.New("no token: lock not held by this instance")
    }
    // 使用 Lua 脚本原子地检查令牌并删除键
    res, err := l.client.Eval(ctx, unlockScript, []string{l.key}, l.token).Result()
    if err != nil {
        return false, err
    }

    // 如果返回值是整数并且等于1，则表示成功删除了键
    if deleted, ok := res.(int64); ok && deleted == 1 {
        l.token = ""
        return true, nil
    }
    return false, nil
}
```

## 添加看门狗
```Go
type Locker struct {
    client *redis.Client
    key    string
    token  string
    ttl    time.Duration
    // cancel 用来管理看门狗的生命周期
    cancel func()
    mu     sync.Mutex
}

  

// NewLocker 创建一个新的 Locker 实例。

func NewLocker(client *redis.Client, key string, ttl time.Duration) *Locker {
    return &Locker{
        client: client,
        key:    fmt.Sprintf("lock:%s", key),
        ttl:    ttl,
    }
}

// genToken 生成一个随机的16字节令牌，用于锁定。
func genToken(nbytes int) (string, error) {
    b := make([]byte, nbytes)
    if _, err := rand.Read(b); err != nil {
        return "", err
    }
    return hex.EncodeToString(b), nil
}

// TryLock 尝试获取锁，返回是否成功和可能的错误。
func (l *Locker) TryLock(ctx context.Context) (bool, error) {
    if l.client == nil {
        return false, errors.New("nil redis client")
    }
    // 生成一个随机令牌
    token, err := genToken(16)
    if err != nil {
        return false, err
    }
    // 使用 SET NX PX 原子操作通过 SetNX 包装器
    ok, err := l.client.SetNX(ctx, l.key, token, l.ttl).Result()
    if err != nil {
        return false, err
    }
    if !ok {
        return false, nil
    }
    l.token = token
    // 成功获取锁后，启动看门狗以定期续期
    l.mu.Lock()
    // 如果之前已经有看门狗在运行，先取消它
    if l.cancel != nil {
        l.cancel()
    }
    wgCtx, cancel := context.WithCancel(context.Background())
    l.cancel = cancel
    l.mu.Unlock()
    // 启动看门狗协程，定期续期锁
    go l.watchDog(wgCtx)
    return true, nil
}

// Lock 是 TryLock 的别名，保持和示例中的调用一致。
func (l *Locker) Lock(ctx context.Context) (bool, error) {
    return l.TryLock(ctx)
}

// unlockScript 解锁脚本，只有当键的值与令牌匹配时才删除键
var unlockScript = `
if redis.call("get", KEYS[1]) == ARGV[1] then
    return redis.call("del", KEYS[1])
else
    return 0
end
`
  
// refreshScript 用于看门狗：只有当令牌匹配时才刷新键的过期时间（毫秒）
var refreshScript = `
if redis.call("get", KEYS[1]) == ARGV[1] then
    return redis.call("pexpire", KEYS[1], ARGV[2])
else
    return 0
end
`

// Unlock 释放锁，返回是否成功和可能的错误。
// 注意：只有持有锁的实例才能释放锁。
func (l *Locker) Unlock(ctx context.Context) (bool, error) {
    if l.client == nil {
        return false, errors.New("nil redis client")
    }
    if l.token == "" {
        return false, errors.New("no token: lock not held by this instance")
    }
    // 取消看门狗协程
    l.mu.Lock()
    if l.cancel != nil {
        l.cancel()
        l.cancel = nil
    }
    l.mu.Unlock()

    // 使用 Lua 脚本原子地检查令牌并删除键
    res, err := l.client.Eval(ctx, unlockScript, []string{l.key}, l.token).Result()
    if err != nil {
        return false, err
    }

    // 如果返回值是整数并且等于1，则表示成功删除了键
    if deleted, ok := res.(int64); ok && deleted == 1 {
        l.token = ""
        return true, nil
    }
    return false, nil
}

// watchDog 定期续期锁的 TTL，直到 ctx 被取消或续期失败。
func (l *Locker) watchDog(ctx context.Context) {
    // 续期间隔：取 ttl 的三分之一，但至少 1s
    interval := l.ttl / 3
    if interval < time.Second {
        interval = time.Second
    }
    ticker := time.NewTicker(interval)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            // 尝试续期：只有当键仍由我们持有（token 匹配）时才刷新过期时间
            // 使用 pexpire（毫秒）以更精确控制
            ttlMillis := l.ttl.Milliseconds()
            _, _ = l.client.Eval(ctx, refreshScript, []string{l.key}, l.token, ttlMillis).Result()
            // 忽略错误：若续期失败（比如键不存在或被取走），下次循环会继续尝试，
            // 但持有者应根据业务决定是否需要强制停止。
        }
    }
}
```

### 存在的问题
主从一致性问题：
目前的实现在主从架构或哨兵模式下，无法解决主从一致性问题

失效场景：
假设你有一个 Redis 集群，采用 **主从复制 (Master-Slave Replication)** 架构：
- **Master (主节点)**：负责写操作（加锁）。
- **Slave (从节点)**：负责备份数据，Master 会异步地把数据同步给 Slave。
- **Sentinel (哨兵)**：负责监控，如果 Master 挂了，自动把 Slave 提升为新 Master。

#### 详细步骤演示
##### 1. 正常加锁
**客户端 A** 发起加锁请求：`SET lock:resource token_A NX PX 10000`。
- 请求发送给 **Master**。
- **Master** 执行成功，内存中有了 `lock:resource`。
- **Master** 返回 "OK" 给客户端 A。
- **此时，客户端 A 认为自己已经拿到了锁，开始执行关键业务逻辑（比如操作数据库、扣钱等）。**

##### 2. 异步复制的时间差 (关键点)
Redis 的主从复制是**异步**的（Asynchronous）。  
这意味着：Master 返回 "OK" 给客户端 A 的时候，它**不保证**这条数据已经传给了 Slave。
- Master 刚给客户端 A 回复完，正准备把 `SET lock:resource ...` 这条命令发给 Slave。

##### 3. 灾难发生 (Master 宕机)
就在这个极其短暂的毫秒级时间窗口内（数据还没传给 Slave），**Master 突然宕机了**（断电、进程崩溃、网络隔离等）。
- **现状**：
    - 客户端 A：持有锁，正在欢快地跑业务。
    - 原 Master：挂了，数据死在了它的内存里。
    - Slave：活得好好的，但它**根本不知道** `lock:resource` 这回事（因为还没同步过来）。

##### 4. 故障转移 (Failover)
**Sentinel (哨兵)** 检测到 Master 挂了。
- 哨兵发起选举，将 **Slave** 提升为 **新 Master**。
- 现在，集群恢复正常了，但数据状态回滚到了 Master 宕机前的一瞬间。

##### 5. 锁冲突 (互斥失效)
**客户端 B** 此时也想来加锁，发起请求：`SET lock:resource token_B NX PX 10000`。
- 请求发送给 **新 Master**。
- **新 Master** 查了一下自己的内存：“嗯，`lock:resource` 不存在，可以加锁！”
- **新 Master** 执行成功，返回 "OK" 给客户端 B。
- **此时，客户端 B 也认为自己拿到了锁，开始执行关键业务逻辑。**

#### 最终后果
- **客户端 A** 正在操作共享资源。
- **客户端 B** 同时也正在操作同一个共享资源。
- **互斥性 (Mutual Exclusion)** 被彻底打破。
