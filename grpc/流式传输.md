## 简单流式
一问一答式：客户端请求一次，服务端响应一次
```Go
syntax = "proto3";  
package stream_proto;  
option go_package = "/stream_proto";  
  
service SimpleStreamService {  
    // 简单流式  
    rpc SimpleStream (Request) returns (Response) {}  
}  
  
message Request {  
    string name = 1;  
}  
  
message Response {  
    string text = 1;  
}
```

## 服务端流式
客户端请求一次，服务端发送多次数据
```Go
syntax = "proto3";  
package stream_proto;  
option go_package = "/stream_proto";  
  
message Request {  
    string name = 1;  
}  
  
message Response {  
    string text = 1;  
}  
  
service ServerStreamService {  
    // 服务器流式  
    rpc ServerStream (Request) returns (stream Response) {}  
}
```
服务端：
```Go
package main  
  
import (  
    "fmt"  
    "google.golang.org/grpc"    
    "grpc_study/stream_proto/stream_proto"    
    "net")  
  
type serverStream struct {  
    // Add fields here  
}  
  
func (serverStream) ServerStream(request *stream_proto.Request, stream stream_proto.ServerStreamService_ServerStreamServer) error {  
    fmt.Println(request)  
    for i := 0; i < 10; i++ {  
       if err := stream.Send(&stream_proto.Response{  
          Text: fmt.Sprintf("server stream %d", i),  
       }); err != nil {  
          return err  
       }  
    }  
    return nil  
}  
  
func main() {  
    listen, err := net.Listen("tcp", ":8081")  
    if err != nil {  
       panic(err)  
    }  
  
    s := grpc.NewServer()  
    stream_proto.RegisterServerStreamServiceServer(s, &serverStream{})  
    if err := s.Serve(listen); err != nil {  
       panic(err)  
    }  
}
```
客户端：
```Go
package main  
  
import (  
    "context"  
    "google.golang.org/grpc"    
    "google.golang.org/grpc/credentials/insecure"    
    "grpc_study/stream_proto/stream_proto")  
  
func main() {  
    addr := ":8081"  
  
    conn, err := grpc.Dial(addr, grpc.WithTransportCredentials(insecure.NewCredentials()))  
    if err != nil {  
       panic(err)  
    }  
    defer conn.Close()  
  
    client := stream_proto.NewServerStreamServiceClient(conn)  
    stream, err := client.ServerStream(context.Background(), &stream_proto.Request{  
       Name: "张三",  
    })  
    if err != nil {  
       panic(err)  
    }  
    for i := 0; i < 10; i++ {  
       resp, err := stream.Recv()  
       if err != nil {  
          panic(err)  
       }  
       println(resp.Text)  
    }  
}
```

### 下载文件
```GO
syntax = "proto3";  
package stream_proto;  
option go_package = "/stream_proto";  
  
message Request {  
    string name = 1;  
}    
  
service ServerStreamService {  
    // 服务器流式   
    rpc DownloadFile (Request) returns (stream FileResponse) {}  
}  
  
message FileResponse {  
    string filename = 1;  
    bytes content = 2;  
}
```

服务端：
```Go
package main  
  
import (  
    "fmt"  
    "google.golang.org/grpc"    
    "grpc_study/stream_proto/stream_proto"    
    "io"    
    "net"    
    "os")  
  
type serverStream struct {  
    // Add fields here  
}  
  
func (serverStream) DownloadFile(request *stream_proto.Request, stream stream_proto.ServerStreamService_DownloadFileServer) error {  
    fmt.Println(request)  
    file, err := os.Open("./static/video.mp4")  
    if err != nil {  
       panic(err)  
    }  
    defer file.Close()  
  
    for {  
       buf := make([]byte, 32)  
       _, err := file.Read(buf)  
       if err == io.EOF {  
          break  
       }  
       if err != nil {  
          return err  
       }  
       stream.Send(&stream_proto.FileResponse{  
          Filename: "video.mp4",  
          Content:  buf,  
       })  
    }  
    return nil  
}  
  
func main() {  
    listen, err := net.Listen("tcp", ":8081")  
    if err != nil {  
       panic(err)  
    }  
  
    s := grpc.NewServer()  
    stream_proto.RegisterServerStreamServiceServer(s, &serverStream{})  
    if err := s.Serve(listen); err != nil {  
       panic(err)  
    }  
}
```

客户端：
```Go
package main  
  
import (  
    "bufio"  
    "context"    
    "google.golang.org/grpc"    
    "google.golang.org/grpc/credentials/insecure"    
    "grpc_study/stream_proto/stream_proto"    
    "io"    
    "os")  
  
func main() {  
    addr := ":8081"  
  
    conn, err := grpc.Dial(addr, grpc.WithTransportCredentials(insecure.NewCredentials()))  
    if err != nil {  
       panic(err)  
    }  
    defer conn.Close()  
  
    client := stream_proto.NewServerStreamServiceClient(conn)  
    stream, err := client.DownloadFile(context.Background(), &stream_proto.Request{  
       Name: "张三",  
    })  
  
    file, err := os.OpenFile("./static/video_copy.mp4", os.O_CREATE|os.O_WRONLY, 0644)  
    if err != nil {  
       panic(err)  
    }  
    defer file.Close()  
  
    writer := bufio.NewWriter(file)  
    for {  
       resp, err := stream.Recv()  
       if err == io.EOF {  
          break  
       }  
       writer.Write(resp.Content)  
    }  
    writer.Flush()  
}
```

```Go
service ClientStreamService {

}

    string filename = 1;  
    bytes content = 2;  
}
```

### 上传文件
服务端：
```Go
package main  
  
import (  
    "bufio"  
    "google.golang.org/grpc"    
    "grpc_study/stream_proto/stream_proto"    
    "io"    
    "net"    
    "os")

}
```

## 双向流传输
```Go
syntax = "proto3";  
package stream_proto;  
option go_package = "/stream_proto";  
  
service BothStreamService {  
    // 双向流式  
    rpc Chat (stream MsgRequest) returns (stream MsgResponse) {}  
}  
  
message MsgRequest {  
    string message = 1;  
}  
  
message MsgResponse {  
    string message = 1;  
}
```

服务端：
```Go
package main  
  
import (  
    "fmt"  
    "google.golang.org/grpc"    
    "grpc_study/stream_proto/stream_proto"    
    "net")  
  
type BothStream struct {  
}  
  
func (BothStream) Chat(stream stream_proto.BothStreamService_ChatServer) error {  
    for {  
       req, err := stream.Recv()  
       if err != nil {  
          return err  
       }  
       fmt.Println(req.Message)  
       stream.Send(&stream_proto.MsgResponse{  
          Message: "hello too",  
       })  
    }  
    return nil  
}  
  
func main() {  
    listen, err := net.Listen("tcp", ":8081")  
    if err != nil {  
       panic(err)  
    }  
  
    s := grpc.NewServer()  
    stream_proto.RegisterBothStreamServiceServer(s, &BothStream{})  
    if err := s.Serve(listen); err != nil {  
       panic(err)  
    }  
}
```

客户端：
